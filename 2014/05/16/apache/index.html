Apache 笔记

Mac下的Apache

sudo apachectl -v  # 查看apache版本
sudo apachectl start  # 启动
sudo apachectl stop  # 停止

虚拟主机配置示例

# 修改http.conf
Listen 8092 # 增加端口监控
Include /private/etc/apache2/other/*.conf # 增加配置文件


# 在other目录下建一个虚拟主机的配置文件 如：test1.conf
NameVirtualHost *:8092
<VirtualHost *:8092>
    ServerAdmin test@163.com
    DocumentRoot "/Users/path-of-web-content"
    ServerName localhost
    ServerAlias localhost
    ErrorLog "/private/var/log/apache2/log-name-error_log"
    CustomLog "/private/var/log/apache2/log-name-access_log" common
    <Directory "/Users/path-of-web-content">
        Options None
        AllowOverride None
        Order allow,deny
        Allow from all
    </Directory>
</VirtualHost>
